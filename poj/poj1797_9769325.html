void Dijkstra()
{
    memset(v, 0, sizeof(v));
    for(int i = 0; i <=n;i++)g[i][i]=INF;
    for(int i = 1; i<=n;i++)
    {
        d[i]=g[1][i];
    }
    for(int i = 0; i < n; i++)
    {
        int x, m = 0;
        for(int y = 1; y<=n;y++)
        {
            if(!v[y] && d[y]>=m)
                m = d[x=y];
        }
        v[x]=1;
        for(int y = 1; y<=n;y++)
        {
            if( !v[y]&&d[y]<min(d[x],g[x][y]))

        }
    }
