int inversion(char str[])
{
	int length = strlen( str);
	int count = 0;
	int i, j;
	for(i =0; i < length; i++)
	{
		for(j =i+1; j < length; j++)
		{
			if(str[j] < str[i])
			{
				count++;
			}
		}
	}

	return count;
