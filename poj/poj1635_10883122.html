string DFS(string str)
{
 if(str == "01")
  return str;
 else
 {
  str = str.substr(1, str.length()-2);
  int i, cnt = 0, start, len = str.length();
  vector< string > sv;
  for(start=i=0; i<len; i++)
  {
   cnt += str[i] == '0' ? 1 : -1;
   if(!cnt)
   {
    sv.push_back(DFS(str.substr(start, i-start+1)));
    start = i + 1;
   }
  }
  sort(sv.begin(), sv.end());

  string str = "0";
  len = sv.size();
  for(i=0; i<len; i++)
  {
   str += sv[i];
  }
  return str + "1";
 }
}
