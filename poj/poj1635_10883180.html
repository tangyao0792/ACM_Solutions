string MinTreeRep(string tree)
{
    if(tree=="01")
        return tree;
	tree=tree.substr(1,tree.size()-2);

	vector<string> v;

	int size=tree.size();
	int last=0;
	int tmp=0;
	for(int i=0;i<size;i++)
	{
		if(tree[i]=='0')
			tmp++;
		if(tree[i]=='1')
			tmp--;
		if(tmp==0)
		{
			v.push_back(MinTreeRep(tree.substr(last,i-last+1)));
			last=i+1;
		}
	}
	sort(v.begin(),v.end());
	tree="";
	size=v.size();
	for(int i=0;i<size;i++)
        tree=tree+v[i];

	return "0"+tree+"1";
}
