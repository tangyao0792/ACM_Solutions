const int MAXN=200010;

struct Tree
{
    int l,r,len;
}T[MAXN*5];

int n,arr[MAXN],p[MAXN],v[MAXN];

void build(int s,int t,int pos)
{
    T[pos].l=s;
    T[pos].r=t;
    T[pos].len=t-s+1;

    if(s==t)    return;

    int mid=(s+t)>>1;

    build(s,mid,pos*2);
    build(mid+1,t,pos*2+1);
}

int find(int p,int pos)
{
    T[pos].len--;
    if(T[pos].l==T[pos].r)
        return T[pos].l;
    if(T[pos*2].len<p)
    {
        return find(p-T[pos*2].len,pos*2+1);
    }

    return    find(p,pos*2);
}
inline void readint(int &ret)
{
	char c;
	do {
	    	c = getchar();
	} while(c < '0' || c > '9');

	ret = c - '0';
	while((c=getchar()) >= '0' && c <= '9')
		ret = ret * 10 + ( c - '0' );
}
