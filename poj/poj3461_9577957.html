void make_next()
{
    next[0] = -1;
    int i =1;
    int j = -1;
    while(word[i-1])
    {
        if(j == -1 || word[i-1] == word[j])
        {
            i++;
            j++;
            if( word[i-1] != word[j])
            {
                next[i-1] = j;
            }
            else
            {
                next[i-1] = next[j];

        {

        }
    }
